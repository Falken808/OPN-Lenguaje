Sintaxis de OPN

Guia rapida de comentarios, tipos, control de flujo, funciones, clases y colecciones.

Comentarios y variables

// Comentario simple
/* Comentario multilinea */
var nombre = "OPN";
var edad = 25;
var activo = true;
var lista = [1, 2, 3];
var mapa = {"a": 1};

Operadores

  • Aritmeticos: +, -, *, /, %
  • Comparacion: ==, !=, >, <, >=, <=
  • Logicos: &&, ||, !

Condicionales y bucles

if (edad < 18) {
  print("Menor");
} else if (edad < 65) {
  print("Adulto");
} else {
  print("Jubilado");
}

for (var i = 0; i < 5; i = i + 1) {
  print(i);
}

Funciones y clases

function suma(a, b) {
  return a + b;
}

class Persona {
  function init(nombre) { this.nombre = nombre; }
  function saludar() { return "Hola, " + this.nombre; }
}
